## 概念
线性表（linear_list）：

>由n个 相同数据元素 组成的 有限序列。 序列：元素之间是有序的，若元素存在多个，则第一个元素无前驱，最后一个元素无后继，其他每个元素都有且只有一个前驱和后继，即相邻元素间是一对一关系  
**有限**：元素的数量是有限的，不过计算机中的对象都是有限的，无限数列只存在于数学概念中

线性表的元素的个数为n（n>=0），n即是线性表的长度，n=0时，即为空表。在非空表中，元素按照顺序的位置（从1开始到第i个）即为元素的位序。

线性表在日常生活中的案例：幼儿园小朋友手拉手排队过马路：每个人记着自己的前一位和后一位。

在复杂的线性表中，一个数据元素有多干数据项（item）组成，此时常把数据元素称为记录（record），含有大量记录的线性表又称为文件（file）。比如上述案例中，每个小朋友都其实是一个数据元素，但是这个小朋友本身也拥有很多自己的属性：名字、年龄、性别等。

>注意：同一个线性表中的元素必定有相同的特性，即属于同一数据对象，且相邻数据元素之间存在序偶关系。

## 划分
按照物理存储结构划分，线性表分为:

+ 顺序表：顺序结构存储。
+ 链表：链式结构存储。

链表的逻辑结构具体实现有两种方式：

+ 动态链表：即常见的单链表、双向链表、循环链表的写法
+ 静态链表：底层不使用指针，而是使用数组实现结点的增删改查

## 实现
下面主要介绍了3种实现：

+ 顺序表
+ 单向链表
+ 循环链表

	s.Arr[index - 1] = e
	s.Length ++

	// 位序越界
	if index < 1 || index > s.Length {
		fmt.Println("位序越界")
		return errors.New("位序越界")
	}
	
	for i := index; i <= s.Length; i ++ {
		if i == s.Length {
			s.Arr[i] = nil
		} else {
			s.Arr[i] = s.Arr[i + 1]
		}
	}

	s.Length --
	return nil

	if index < 1 || index > s.Length {

// Locate 寻找元素为e的下标
func (s *SqList) Locate(e interface{}) (index int, err error) {
	for i := 0; i < s.Length; i ++ {
		if s.Arr[i] == e {
			return i + 1, nil
		}
	}
	return -1, errors.New("未找到数据")
}

// PrevElem 寻找元素为e的前一个元素
func (s *SqList) PrevElem(e interface{}) (interface{}, error) {
	if s.Length == 0 {
		return nil, errors.New("数据结构为空")
	}

	index, err := s.Locate(e)
	if err != nil {
		return nil, err
	}

	if index == 1 {
		fmt.Println("第一个元素无前驱")
		return nil, errors.New("第一个元素无前驱")
	}

	return s.Arr[(index - 1) - 1], nil
}

// NextElem 寻找元素为e的后一个元素
func (s *SqList) NextElem(e interface{}) (interface{}, error) {
	if s.Length == 0 {
		return nil, errors.New("数据结构为空")
	}

	index, err := s.Locate(e)
	if err != nil {
		return nil, err
	}

	if index == s.Length {
		fmt.Println("最后一个元素无后继")
		return nil, errors.New("最后一个元素无后继")
	}

	return s.Arr[(index - 1) + 1], nil
}

// Clear 清空
func (s *SqList) Clear() {
	s.Length = 0
	s.Arr = make([]interface{}, s.Size)
}

package LinkList

import (
	"errors"
	"fmt"
)

// Node 单节点
type Node struct {
	Data interface{}		// 数据
	Next *Node				// 下一节点
}

type LinkList struct {
	Length int
	Head *Node				// 头节点
}

func NewLinkList() *LinkList {
	return &LinkList{
		Length: 0,
		Head:   nil,
	}
}

func (l *LinkList) GetLength() int {
	return l.Length
}

func (l *LinkList) Display()  {
	if l.Length == 0 {
		fmt.Println("数据结构为空")
		return
	}

	fmt.Printf("数据元素显示：")
	currentNode := l.Head
	for currentNode.Next != nil {
		fmt.Printf("%d ", currentNode.Data)
		currentNode = currentNode.Next
	}
	fmt.Printf("%d ", currentNode.Data)
	fmt.Println("")
}

func (l *LinkList) Append(e interface{}) {
	// 构造要插入的节点
	newNode := &Node{
		Data: e,
		Next: nil,
	}

	// 当前循环到的节点
	currentNode := l.Head

	// 第一次追加
	if currentNode == nil {
		l.Head = newNode
		l.Length ++
		return
	}

	// 常见追加情况
	for currentNode.Next != nil {
		currentNode = currentNode.Next
	}
	currentNode.Next = newNode
	l.Length++
}

func (l *LinkList) Insert(index int, e interface{}) error {
	if index < 1 || index > l.Length {
		fmt.Println("位序非法")
		return errors.New("位序非法")
	}
	
	insertNode := &Node{
		Data: e,
		Next: nil,
	}

	currentNode := l.Head

	// 如果是在第一个节点插入
	if index == 1 {
		l.Head = insertNode
		l.Head.Next = currentNode
		l.Length ++
		return nil
	}

	// 常见插入：找到插入位置的前一个节点
	i := 1
	for currentNode.Next != nil {
		if i == index - 1 {
			break
		}
		i ++
		currentNode = currentNode.Next
	}

	// 执行插入
	insertNode.Next = currentNode.Next
	currentNode.Next = insertNode
	l.Length ++
	return nil
}

func (l *LinkList) Delete(index int) error {
	if index < 1 || index > l.Length {
		fmt.Println("位序非法")
		return errors.New("位序非法")
	}

	if l.Length == 0 {
		fmt.Println("数据结构为空")
		return errors.New("数据结构为空")
	}

	// 如果删除的是第一个元素
	if index == 1 {
		if l.Length == 1 {
			l.Head = nil
		} else {
			l.Head = l.Head.Next
		}
		l.Length --
		return nil
	}

	// 常规删除：找到要删除元素的前一个元素
	i := 1
	currentNode := l.Head
	for currentNode.Next != nil {
		if i == index - 1 {
			break
		}
		i ++
		currentNode = currentNode.Next
	}
	currentNode.Next = currentNode.Next.Next
	l.Length --
	return nil
}

func (l *LinkList) Update(index int, e interface{}) error {
	if index < 1 || index > l.Length {
		fmt.Println("位序非法")
		return errors.New("位序非法")
	}

	i := 1
	currentNode := l.Head
	for currentNode.Next != nil {
		if i == index {
			break
		}
		i ++
		currentNode = currentNode.Next
	}

	currentNode.Data = e
	return nil
}

func (l *LinkList) GetElem(index int) (e interface{}, err error) {
	if index < 1 || index > l.Length {
		fmt.Println("位序非法")
		return nil, errors.New("位序非法")
	}

	i := 1
	currentNode := l.Head
	for currentNode.Next != nil {
		if i == index {
			break
		}
		i ++
		currentNode = currentNode.Next
	}

	return currentNode.Data, nil
}

func (l *LinkList) Locate(e interface{}) (index int, err error) {
	i := 1
	currentNode := l.Head
	for currentNode.Next != nil {
		if currentNode.Data == e {
			break
		}
		i++
		currentNode = currentNode.Next
	}

	if i == l.Length && currentNode.Data != e {
		fmt.Println("未找到元素")
		return -1, errors.New("未找到元素")
	}

	return i, nil
}

func (l *LinkList) PrevElem(e interface{}) (pe interface{}, err error) {
	if l.Length <= 1 {
		fmt.Println("数据结构为空")
		err = errors.New("数据结构为空")
		return
	}

	if l.Head.Data == e {
		fmt.Println("首元素无前驱")
		err = errors.New("首元素无前驱")
		return
	}

	i := 1
	currentNode := l.Head
	for currentNode.Next != nil {
		if currentNode.Next.Data == e {
			pe = currentNode.Data
			return
		}
		i++
		currentNode = currentNode.Next
	}

	fmt.Println("元素未找到")
	err = errors.New("元素未找到")
	return
}

func (l *LinkList) NextElem(e interface{}) (ne interface{}, err error) {
	if l.Length <= 1 {
		fmt.Println("数据结构为空")
		err = errors.New("数据结构为空")
		return
	}

	i := 1
	currentNode := l.Head
	for currentNode.Next != nil {
		if currentNode.Data == e {
			break
		}
		i ++
		currentNode = currentNode.Next
	}

	if i == l.Length && currentNode.Data != e {
		fmt.Println("元素未找到")
		err = errors.New("元素未找到")
		return
	}

	if i == l.Length && currentNode.Data == e {
		fmt.Println("最后元素无后继")
		err = errors.New("最后元素无后继")
		return
	}

	ne = currentNode.Next.Data
	return
}

func (l *LinkList) Clear() {
	l.Head = nil
	l.Length = 0
}

package DcList

import (
	"errors"
	"fmt"
)

type Node struct {
	data interface{}
	next *Node
	prev *Node
}

type DcList struct {
	length int
	head   *Node
}

func NewDcList() *DcList {
	return &DcList{
		length: 0,
		head:   nil,
	}
}

func (d *DcList) Length() int {
	return d.length
}

func (d *DcList) Display() {
	if d.length == 0 {
		fmt.Println("数据结构内无元素")
		return
	}

	fmt.Printf("显示表：")
	currentNode := d.head
	for currentNode.next != d.head {
		fmt.Printf("%d ", currentNode.data)
		currentNode = currentNode.next
	}
	fmt.Printf("%d ", currentNode.data)
	fmt.Println()
}

func (d *DcList) Append(e interface{}) {
	insertNode := &Node{
		data: e,
		next: nil,
		prev: nil,
	}

	// 第一次追加
	if d.length == 0 {
		insertNode.next = insertNode
		insertNode.prev = insertNode
		d.head = insertNode
		d.length++
		return
	}

	// 常规追加：找到最后一个元素
	currentNode := d.head
	for currentNode.next != d.head {
		currentNode = currentNode.next
	}

	// 设置insertNode
	insertNode.next = d.head
	insertNode.prev = currentNode

	// 执行追加
	currentNode.next = insertNode
	d.head.prev = insertNode
	d.length ++
}

func (d *DcList) Insert(index int, e interface{}) error {
	if index < 1 || index > d.length + 1 {
		fmt.Println("位序非法")
		return errors.New("位序非法")
	}

	// 如果是末尾插入
	if d.length == 0 || d.length + 1 == index {
		d.Append(e)
		return nil
	}

	// 常规插入：构造要插入的节点
	insertNode := &Node{
		data: e,
		next: nil,
		prev: nil,
	}

	// 查找插入位置
	i := 1
	currentNode := d.head

	for currentNode.next != d.head {
		if index == i {
			break
		}
		i ++
		currentNode = currentNode.next
	}

	// 设置insertNode
	insertNode.prev = currentNode.prev
	insertNode.next = currentNode

	// 执行插入
	currentNode.prev.next = insertNode
	currentNode.prev = insertNode
	d.length ++

	// 考虑头指针是否改变
	if index == 1 {
		d.head = insertNode
	}

	return nil
}

func (d *DcList) Delete(index int) error {
	if index < 1 || index > d.length + 1 {
		fmt.Println("位序非法")
		return errors.New("位序非法")
	}

	// 如果删除的是第一个元素
	if index == 1 && d.length == 1 {
		d.head = nil
		d.length = 0
		return nil
	}

	// 找到要删除元素的元素
	i := 1
	currentNode := d.head

	for currentNode.next != d.head {
		if i == index {
			break
		}
		i ++
		currentNode = currentNode.next
	}

	// 执行删除
	prevNode := currentNode.prev
	nextNode := currentNode.next
	prevNode.next = nextNode
	nextNode.prev = prevNode
	d.length --

	// 考虑头指针是否会改变
	if index == 1 {
		d.head = nextNode
	}

	return nil
}

func (d *DcList) Update(index int, e interface{}) error {
	if index < 1 || index > d.length + 1 {
		fmt.Println("位序非法")
		return errors.New("位序非法")
	}

	i := 1
	currentNode := d.head

	for currentNode.next != d.head {
		if i == index {
			break
		}
		i ++
		currentNode = currentNode.next
	}

	currentNode.data = e
	return nil
}

func (d *DcList) GetElem(index int) (e interface{}, err error) {
	if index < 1 || index > d.length + 1 {
		fmt.Println("位序非法")
		return nil, errors.New("位序非法")
	}

	i := 1
	currentNode := d.head
	for currentNode.next != d.head {
		if i == index {
			break
		}
		i ++
		currentNode = currentNode.next
	}

	return currentNode.data, nil
}

func (d *DcList) Locate(e interface{}) (index int, err error) {
	i := 1
	currentNode := d.head
	for currentNode.next != d.head {
		if currentNode.data == e {
			break
		}
		i ++
		currentNode = currentNode.next
	}

	if i == d.length && currentNode.data != e {
		fmt.Println("未找到元素")
		err = errors.New("未找到元素")
		return
	}

	index = i
	return
}

func (d *DcList) PrevElem(e interface{}) (pe interface{}, err error) {
	if d.length <= 1 {
		fmt.Println("数据结构元素不足，无法查询")
		err = errors.New("数据结构元素不足，无法查询")
		return
	}

	if d.head.data == e {
		fmt.Println("首元素无前驱")
		err = errors.New("首元素无前驱")
		return
	}

	i := 1
	currentNode := d.head
	for currentNode.next != d.head {
		if currentNode.data == e {
			pe = currentNode.data
			return
		}
		i ++
		currentNode = currentNode.next
	}
	fmt.Println("数据结构元素不足，无法查询")
	err = errors.New("未找到传入元素")
	return
}

func (d *DcList)NextElem(e interface{}) (ne interface{}, err error) {
	if d.length <= 1 {
		fmt.Println("数据结构元素不足，无法查询")
		err = errors.New("数据结构元素不足，无法查询")
		return
	}

	i := 1
	currentNode := d.head
	for currentNode.next != d.head {
		if currentNode.data == e {
			break
		}
		i ++
		currentNode = currentNode.next
	}

	if i == d.length && currentNode.data != e {
		fmt.Println("未找到传入元素")
		err = errors.New("未找到传入元素")
		return
	}

	if i == d.length && currentNode.data == e {
		fmt.Println("末尾元素无后继")
		err = errors.New("末尾元素无后继")
		return
	}

	ne = currentNode.next.data
	return
}

func (d *DcList) Clear()  {
	d.head = nil
	d.length = 0
}
